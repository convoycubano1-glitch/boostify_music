rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Global rule - allow authenticated users to read
    match /{allPaths=**} {
      allow read: if true; // Permitir lectura pública para facilitar compartir contenido
    }
    
    // Allow authenticated users to upload music files
    match /artist_music/{userId}/{songId} {
      allow read: if true; // Permitir lectura pública para facilitar compartir contenido
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 50 * 1024 * 1024 // 50MB max
                   && (request.resource.contentType.matches('audio/.*') || // Allow all audio types
                       request.resource.contentType.matches('audio/mpeg') || // MP3
                       request.resource.contentType.matches('audio/wav') || // WAV
                       request.resource.contentType.matches('audio/x-wav') || // WAV alternative
                       request.resource.contentType.matches('audio/ogg') || // OGG
                       request.resource.contentType.matches('audio/aac') || // AAC
                       request.resource.contentType.matches('audio/webm')); // WebM audio
    }
    
    // Allow authenticated users to upload videos
    match /videos/{category}/{videoId} {
      allow read: if true; // Permitir lectura pública para facilitar compartir contenido
      allow write: if request.auth != null
                  && request.resource.size < 200 * 1024 * 1024 // 200MB max for videos
                  && (request.resource.contentType.matches('video/.*')); // Allow all video types
    }
    
    // Allow image uploads for thumbnails and user profiles
    match /images/{userId}/{imageId} {
      allow read: if true; // Permitir lectura pública para facilitar compartir contenido
      allow write: if request.auth != null 
                  && request.resource.size < 5 * 1024 * 1024 // 5MB max for images
                  && (request.resource.contentType.matches('image/.*'));
    }
  }
}
