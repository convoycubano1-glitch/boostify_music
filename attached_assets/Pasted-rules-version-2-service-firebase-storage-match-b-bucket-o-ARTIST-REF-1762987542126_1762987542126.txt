rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    
    // ✅ ARTIST REFERENCES - PERMITIR SUBIR IMÁGENES DE REFERENCIA
    match /artist-references/{userId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024  // 10MB max
                   && request.resource.contentType.matches('image/.*');
    }
    
    // ✅ ARTIST PROFILE IMAGES - GEMINI GENERATED
    match /artist-profile-images/{userId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }
    
    // ✅ ARTIST BANNER IMAGES - GEMINI GENERATED
    match /artist-banner-images/{userId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }
    
    // ✅ GENERATED IMAGES - AI GENERATED
    match /generated-images/{userId}/{fileName} {
      allow read: if true;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 10 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }
    
    // Artist music files
    match /artist_music/{userId}/{songId} {
      allow read: if true;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 50 * 1024 * 1024  // 50MB max
                   && (request.resource.contentType.matches('audio/.*') ||
                       request.resource.contentType.matches('audio/mpeg') ||
                       request.resource.contentType.matches('audio/wav') ||
                       request.resource.contentType.matches('audio/x-wav') ||
                       request.resource.contentType.matches('audio/ogg') ||
                       request.resource.contentType.matches('audio/aac') ||
                       request.resource.contentType.matches('audio/webm'));
    }
    
    // Videos
    match /videos/{category}/{videoId} {
      allow read: if true;
      allow write: if request.auth != null
                  && request.resource.size < 200 * 1024 * 1024  // 200MB max
                  && request.resource.contentType.matches('video/.*');
    }
    
    // Images - general
    match /images/{userId}/{imageId} {
      allow read: if true;
      allow write: if request.auth != null 
                  && request.resource.size < 5 * 1024 * 1024  // 5MB max
                  && request.resource.contentType.matches('image/.*');
    }
    
    // ✅ FALLBACK - PERMITIR TODO A USUARIOS AUTENTICADOS
    match /{allPaths=**} {
      allow read: if true;  // Lectura pública
      allow write: if request.auth != null;  // Escritura solo autenticados
    }
  }
}
